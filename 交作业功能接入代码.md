# 「交作业」功能接入说明

共需修改 **3 个地方**，按顺序操作即可。

---

## 第一步：添加 State 变量

在 `ClassPointSystem` 组件的 state 声明区域（其他 `useState` 附近）加入以下三行：

```js
// 交作业面板相关状态（最多同时开3个科目）
const [hwSubmitSlots, setHwSubmitSlots] = useState([null, null, null]); // 每个槽: { subject, submittedIds: [] }
const [hwSubjectPickerSlot, setHwSubjectPickerSlot] = useState(null); // 正在选科目的槽位索引
const [hwConfirmResult, setHwConfirmResult] = useState(null); // { slotIdx, subject, missing, deductPoints, ruleName }
```

---

## 第二步：添加导航栏 Tab 按钮

在导航栏的 tab 按钮列表里，找到「作业明细」按钮，在它**后面**插入以下代码：

```jsx
<button
    onClick={() => setActiveTab('homeworkSubmit')}
    className={`flex items-center gap-2 px-4 py-2 rounded-lg transition ${
        activeTab === 'homeworkSubmit' ? 'bg-teal-500 text-white' : 'text-gray-600 hover:bg-gray-100'
    }`}
>
    <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 5H7a2 2 0 00-2 2v12a2 2 0 002 2h10a2 2 0 002-2V7a2 2 0 00-2-2h-2M9 5a2 2 0 002 2h2a2 2 0 002-2M9 5a2 2 0 012-2h2a2 2 0 012 2m-3 7l3 3 3-3" />
    </svg>
    交作业
</button>
```

---

## 第三步：添加 Tab 内容页面

在 tab 内容渲染区（其他 `activeTab === 'xxx' && (...)` 的同级位置），插入以下完整代码块：

```jsx
{/* 交作业面板 */}
{activeTab === 'homeworkSubmit' && (() => {
    const allSubjects = ['语文', '数学', '英语', '物理', '化学', '历史', '道法', '地理', '生物'];
    const sortedStudents = [...students].sort((a, b) => a.name.localeCompare(b.name, 'zh-CN'));

    // 找到与科目相关的减分规则（未交作业）
    const getSubjectDeductRule = (subject) => {
        // 优先匹配含有科目名+不交/未交的规则
        const rule = pointRules.find(r =>
            r.type === 'subtract' &&
            r.name.includes(subject) &&
            (r.name.includes('不交') || r.name.includes('未交') || r.name.includes('没交') || r.name.includes('作业'))
        );
        if (rule) return rule;
        // 次选：任意未交作业规则
        const fallback = pointRules.find(r =>
            r.type === 'subtract' &&
            (r.name.includes('不交') || r.name.includes('未交') || r.name.includes('没交') || r.name.includes('作业未交'))
        );
        return fallback || null;
    };

    const openSlot = (slotIdx) => {
        setHwSubjectPickerSlot(slotIdx);
    };

    const selectSubject = (subject) => {
        const newSlots = [...hwSubmitSlots];
        newSlots[hwSubjectPickerSlot] = { subject, submittedIds: [] };
        setHwSubmitSlots(newSlots);
        setHwSubjectPickerSlot(null);
    };

    const closeSlot = (slotIdx) => {
        const newSlots = [...hwSubmitSlots];
        newSlots[slotIdx] = null;
        setHwSubmitSlots(newSlots);
        setHwConfirmResult(null);
    };

    const toggleSubmitted = (slotIdx, studentId) => {
        const newSlots = hwSubmitSlots.map((slot, i) => {
            if (i !== slotIdx || !slot) return slot;
            const ids = slot.submittedIds.includes(studentId)
                ? slot.submittedIds.filter(id => id !== studentId)
                : [...slot.submittedIds, studentId];
            return { ...slot, submittedIds: ids };
        });
        setHwSubmitSlots(newSlots);
    };

    const confirmDeduct = (slotIdx) => {
        const slot = hwSubmitSlots[slotIdx];
        if (!slot) return;
        const missingStudents = sortedStudents.filter(s => !slot.submittedIds.includes(s.id));
        if (missingStudents.length === 0) {
            setUploadMessage('✅ ' + slot.subject + '：全班都交了作业！');
            setTimeout(() => setUploadMessage(''), 3000);
            closeSlot(slotIdx);
            return;
        }
        const rule = getSubjectDeductRule(slot.subject);
        const deductPoints = rule ? rule.points : -2;
        const ruleName = rule ? rule.name : (slot.subject + '未交作业');
        const timestamp = Date.now();

        const updatedStudents = students.map(s => {
            if (missingStudents.find(ms => ms.id === s.id)) {
                return {
                    ...s,
                    points: parseInt(s.points) + deductPoints,
                    availablePoints: parseInt(s.availablePoints || 0) + deductPoints
                };
            }
            return s;
        });
        setStudents(updatedStudents);

        const newRecords = missingStudents.map((s, idx) => ({
            id: timestamp + idx,
            studentId: s.id,
            studentName: s.name,
            points: deductPoints,
            rule: ruleName,
            time: new Date().toISOString().replace('T', ' ').substring(0, 19),
            type: 'subtract'
        }));
        setHistory(prev => [...newRecords, ...prev]);

        setOperationHistory(prev => [...prev, ...newRecords.map(h => ({
            type: 'point',
            data: {
                studentId: h.studentId,
                points: deductPoints,
                rule: ruleName,
                timestamp: Date.now(),
                historyId: h.id
            }
        }))]);

        setHwConfirmResult({ slotIdx, subject: slot.subject, missing: missingStudents, deductPoints, ruleName });
    };

    const dismissResult = (slotIdx) => {
        setHwConfirmResult(null);
        closeSlot(slotIdx);
    };

    const usedSubjects = hwSubmitSlots.filter(Boolean).map(s => s.subject);

    return (
        <div>
            <div className="bg-white rounded-xl shadow-lg p-6 mb-6">
                <h2 className="text-2xl font-bold text-gray-800 mb-2 flex items-center gap-3">
                    <svg className="w-7 h-7 text-orange-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 5H7a2 2 0 00-2 2v12a2 2 0 002 2h10a2 2 0 002-2V7a2 2 0 00-2-2h-2M9 5a2 2 0 002 2h2a2 2 0 002-2M9 5a2 2 0 012-2h2a2 2 0 012 2m-3 7l3 3 3-3" />
                    </svg>
                    交作业统计
                </h2>
                <p className="text-gray-500 text-sm">同时开启最多 <strong>3 个科目</strong>，点击已交作业学生的名字将其标记（名字消失），最后剩下的是未交的同学，点击「确认扣分」按规则扣分。</p>
            </div>

            {/* 3个科目槽 */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                {hwSubmitSlots.map((slot, slotIdx) => (
                    <div key={slotIdx} className="bg-white rounded-xl shadow-lg overflow-hidden flex flex-col" style={{minHeight: '420px'}}>
                        {!slot ? (
                            /* 空槽：选择科目 */
                            <div className="flex flex-col items-center justify-center h-full py-16 px-6">
                                <div className="text-5xl mb-4">📚</div>
                                <p className="text-gray-400 text-center mb-4">点击添加一个科目</p>
                                <button
                                    onClick={() => openSlot(slotIdx)}
                                    className="px-6 py-3 bg-gradient-to-r from-teal-500 to-cyan-500 text-white rounded-xl font-semibold hover:from-teal-600 hover:to-cyan-600 transition shadow-md"
                                >
                                    + 选择科目
                                </button>
                            </div>
                        ) : (
                            /* 已激活槽 */
                            <div className="flex flex-col h-full">
                                {/* 头部 */}
                                <div className="bg-gradient-to-r from-orange-500 to-amber-500 px-4 py-3 flex items-center justify-between">
                                    <div className="flex items-center gap-2 text-white">
                                        <span className="text-lg font-bold">{slot.subject}</span>
                                        <span className="text-xs bg-white bg-opacity-30 px-2 py-0.5 rounded-full">
                                            已交 {slot.submittedIds.length}/{sortedStudents.length}
                                        </span>
                                    </div>
                                    <button
                                        onClick={() => closeSlot(slotIdx)}
                                        className="text-white hover:bg-white hover:bg-opacity-20 rounded p-1 transition"
                                        title="关闭此科目"
                                    >
                                        <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
                                        </svg>
                                    </button>
                                </div>

                                {/* 提示 */}
                                <div className="px-3 py-2 bg-amber-50 text-xs text-amber-700 border-b border-amber-100">
                                    📌 点击已交作业同学的名字 → 名字消失；剩余的即为未交
                                </div>

                                {/* 学生名字列表（未交的显示为红色标签） */}
                                <div className="flex-1 overflow-y-auto p-3">
                                    <div className="flex flex-wrap gap-2">
                                        {sortedStudents.map(student => {
                                            const submitted = slot.submittedIds.includes(student.id);
                                            if (submitted) return null; // 已交的隐藏
                                            return (
                                                <button
                                                    key={student.id}
                                                    onClick={() => toggleSubmitted(slotIdx, student.id)}
                                                    className="px-3 py-1.5 bg-red-50 border-2 border-red-200 text-red-700 rounded-lg text-sm font-semibold hover:bg-red-100 transition-all hover:scale-95 active:scale-90"
                                                >
                                                    {student.name}
                                                </button>
                                            );
                                        })}
                                    </div>
                                    {sortedStudents.every(s => slot.submittedIds.includes(s.id)) && (
                                        <div className="text-center py-6 text-green-600 font-semibold">
                                            🎉 全班都交了！
                                        </div>
                                    )}
                                </div>

                                {/* 已提交区（可点击撤销） */}
                                {slot.submittedIds.length > 0 && (
                                    <div className="px-3 py-2 bg-green-50 border-t border-green-100">
                                        <div className="text-xs text-green-600 font-semibold mb-1.5">✅ 已交（点击可撤销）：</div>
                                        <div className="flex flex-wrap gap-1.5">
                                            {slot.submittedIds.map(sid => {
                                                const s = students.find(st => st.id === sid);
                                                if (!s) return null;
                                                return (
                                                    <button
                                                        key={sid}
                                                        onClick={() => toggleSubmitted(slotIdx, sid)}
                                                        className="px-2 py-0.5 bg-green-100 border border-green-300 text-green-700 rounded text-xs font-medium hover:bg-red-50 hover:border-red-300 hover:text-red-600 transition"
                                                        title="点击撤销"
                                                    >
                                                        {s.name}
                                                    </button>
                                                );
                                            })}
                                        </div>
                                    </div>
                                )}

                                {/* 确认扣分按钮 */}
                                <div className="px-3 py-3 border-t border-gray-100 bg-gray-50">
                                    {(() => {
                                        const missing = sortedStudents.filter(s => !slot.submittedIds.includes(s.id));
                                        const rule = getSubjectDeductRule(slot.subject);
                                        const pts = rule ? rule.points : -2;
                                        return (
                                            <div>
                                                <div className="text-xs text-gray-500 mb-2">
                                                    未交：<span className="font-bold text-red-600">{missing.length}</span> 人
                                                    {rule ? (
                                                        <span className="ml-2 text-red-500">→ 每人 {pts} 分（规则：{rule.name}）</span>
                                                    ) : (
                                                        <span className="ml-2 text-orange-500">→ 未找到相关规则，将扣 -2 分</span>
                                                    )}
                                                </div>
                                                <button
                                                    onClick={() => confirmDeduct(slotIdx)}
                                                    disabled={missing.length === 0}
                                                    className={`w-full py-2 rounded-lg font-bold text-sm transition ${
                                                        missing.length === 0
                                                            ? 'bg-gray-200 text-gray-400 cursor-not-allowed'
                                                            : 'bg-gradient-to-r from-red-500 to-rose-500 text-white hover:from-red-600 hover:to-rose-600 shadow-md'
                                                    }`}
                                                >
                                                    {missing.length === 0 ? '✅ 全班已交' : `确认扣分（${missing.length} 人各 ${pts} 分）`}
                                                </button>
                                            </div>
                                        );
                                    })()}
                                </div>
                            </div>
                        )}
                    </div>
                ))}
            </div>

            {/* 科目选择弹窗 */}
            {hwSubjectPickerSlot !== null && (
                <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-center justify-center p-4">
                    <div className="bg-white rounded-2xl shadow-2xl p-6 w-full max-w-sm">
                        <h3 className="text-xl font-bold text-gray-800 mb-4">选择科目</h3>
                        <div className="grid grid-cols-3 gap-3">
                            {allSubjects.map(sub => {
                                const used = usedSubjects.includes(sub);
                                return (
                                    <button
                                        key={sub}
                                        onClick={() => !used && selectSubject(sub)}
                                        disabled={used}
                                        className={`py-3 rounded-xl font-bold text-base transition ${
                                            used
                                                ? 'bg-gray-100 text-gray-400 cursor-not-allowed'
                                                : 'bg-gradient-to-br from-teal-50 to-cyan-50 border-2 border-teal-200 text-teal-700 hover:from-teal-100 hover:to-cyan-100'
                                        }`}
                                    >
                                        {sub}
                                        {used && <div className="text-xs font-normal">已开启</div>}
                                    </button>
                                );
                            })}
                        </div>
                        <button
                            onClick={() => setHwSubjectPickerSlot(null)}
                            className="mt-4 w-full py-2 bg-gray-200 text-gray-700 rounded-lg hover:bg-gray-300 transition font-semibold"
                        >
                            取消
                        </button>
                    </div>
                </div>
            )}

            {/* 扣分结果弹窗 */}
            {hwConfirmResult && (
                <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-center justify-center p-4">
                    <div className="bg-white rounded-2xl shadow-2xl p-6 w-full max-w-md">
                        <div className="bg-gradient-to-r from-red-500 to-rose-500 -mx-6 -mt-6 px-6 py-5 rounded-t-2xl mb-5">
                            <h3 className="text-xl font-bold text-white">📋 扣分结果</h3>
                            <p className="text-white text-opacity-90 text-sm mt-1">{hwConfirmResult.subject} · {hwConfirmResult.missing.length} 人未交</p>
                        </div>
                        <div className="mb-4">
                            <div className="text-sm text-gray-500 mb-2">
                                规则：<span className="font-semibold text-gray-700">{hwConfirmResult.ruleName}</span>，每人 <span className="font-bold text-red-600">{hwConfirmResult.deductPoints}</span> 分
                            </div>
                            <div className="flex flex-wrap gap-2 max-h-48 overflow-y-auto">
                                {hwConfirmResult.missing.map(s => (
                                    <span key={s.id} className="px-3 py-1 bg-red-50 border border-red-200 text-red-700 rounded-lg text-sm font-semibold">
                                        {s.name}
                                    </span>
                                ))}
                            </div>
                        </div>
                        <button
                            onClick={() => dismissResult(hwConfirmResult.slotIdx)}
                            className="w-full py-3 bg-gradient-to-r from-teal-500 to-cyan-500 text-white rounded-xl font-bold hover:from-teal-600 hover:to-cyan-600 transition"
                        >
                            确认并关闭此科目
                        </button>
                    </div>
                </div>
            )}
        </div>
    );
})()}
```

---

## 依赖说明

以上代码直接使用了原系统中已有的以下变量和函数，无需额外引入：

| 变量/函数 | 用途 |
|---|---|
| `students` | 学生列表 |
| `setStudents` | 更新学生积分 |
| `pointRules` | 积分规则（用于匹配扣分规则） |
| `setHistory` | 写入积分历史 |
| `setOperationHistory` | 写入操作历史（支持撤销） |
| `setUploadMessage` | 显示顶部提示消息 |
| `activeTab` / `setActiveTab` | tab 切换控制 |
